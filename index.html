// Lyricalmyrical Books — Google Sheets webhook (v2)
// Reliable contract: receives { version, eventId, sentAt, payload }
// Idempotent writes: duplicate eventId values are ignored.

function doPost(e) {
  try {
    var body = readBody_(e);
    var req = JSON.parse(body);
    if (!req || !req.payload) throw new Error('Missing payload');

    var eventId = String(req.eventId || Utilities.getUuid());
    var data = req.payload;
    var lock = LockService.getScriptLock();
    lock.waitLock(10000);

    try {
      if (alreadyProcessed_(eventId)) {
        return json_({ ok: true, duplicate: true, eventId: eventId });
      }
      appendEvent_(eventId, data);
      markProcessed_(eventId);
      return json_({ ok: true, eventId: eventId });
    } finally {
      lock.releaseLock();
    }
  } catch (err) {
    return json_({ ok: false, error: err.message });
  }
}

function doGet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheetName = ss ? ss.getName() : 'No sheet bound';
  return json_({ 
    ok: true, 
    service: 'lyrical-sheets-webhook-v2',
    sheetName: sheetName,
    online: true 
  });
}

function appendEvent_(eventId, data) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('Events') || ss.insertSheet('Events');
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['Logged At', 'Event ID', 'Book', 'Type', 'Date', 'Reference', 'Channel/Store', 'Qty', 'Price/Rate', 'Total/Amount Due', 'Stock After', 'Status', 'Notes']);
    sheet.setFrozenRows(1);
  }

  var loggedAt = new Date();
  var type = data.type || '';
  var ref = type === 'order' ? (data.num || '') : (data.event || '');
  var channel = type === 'order' ? (data.chan || '') : (data.store || '');
  var unit = type === 'order' ? Number(data.price || 0) : Number(data.rate || 0);
  var total = type === 'order' ? Number(data.total || 0) : Number(data.amountDue || 0);
  var status = data.status || '';

  sheet.appendRow([
    loggedAt,
    eventId,
    data.book || '',
    type,
    data.date || '',
    ref,
    channel,
    Number(data.qty || 0),
    unit,
    total,
    Number(data.stockAfter || 0),
    status,
    data.notes || ''
  ]);
}

function alreadyProcessed_(eventId) {
  var p = PropertiesService.getScriptProperties();
  return p.getProperty('evt_' + eventId) === '1';
}

function markProcessed_(eventId) {
  PropertiesService.getScriptProperties().setProperty('evt_' + eventId, '1');
}

function readBody_(e) {
  if (e && e.postData && e.postData.contents) return e.postData.contents;
  if (e && e.parameter && e.parameter.payload) return e.parameter.payload;
  throw new Error('No request body');
}

function json_(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
