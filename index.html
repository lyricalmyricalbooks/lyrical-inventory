// Lyricalmyrical Books — Google Sheets sync
// Handles both fetch() no-cors (raw body) and form POST (e.parameter)
function doPost(e) {
  try {
    var raw = '';

    // Try raw body first (from fetch no-cors)
    if (e.postData && e.postData.contents) {
      raw = e.postData.contents;
    }

    // Fall back to form parameter
    if (!raw && e.parameter && e.parameter.payload) {
      raw = e.parameter.payload;
    }

    if (!raw) throw new Error('No data received');

    var data = JSON.parse(raw);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var bookName = (data.book || 'Unknown').replace(/[^a-zA-Z0-9 \-]/g, '');
    var timestamp = new Date().toLocaleString('en-GB');

    var bookShort = {'Un Fantastico Altrove':'Altrove','The Hound':'Hound','Archaeology of Presence':'Archaeology','Sistema_non_autorizzato':'Sistema','As if Nobody is Watching':'Nobody','Collective Photobook':'Collective'};
    var shortName = bookShort[data.book] || bookName;
    var sheet = ss.getSheetByName(shortName) || ss.insertSheet(shortName);

    if (data.type === 'order') {
      // Find or create the ORDERS section header
      var ordersHeaderRow = -1;
      var lastRow = sheet.getLastRow();
      for (var r = 1; r <= lastRow; r++) {
        var val = sheet.getRange(r, 1).getValue();
        if (val === 'ORDERS') { ordersHeaderRow = r; break; }
      }
      if (ordersHeaderRow === -1) {
        // Build the full structure if sheet is new
        sheet.getRange(1,1).setValue(data.book);
        sheet.getRange(1,1).setFontWeight('bold').setFontSize(13)
          .setBackground('#'+({'Altrove':'C8913A','Hound':'3A7CC8','Archaeology':'7A5C3A','Sistema':'2A7A5C','Nobody':'8A3A7A','Collective':'C8913A'}[shortName]||'C8913A'))
          .setFontColor('#F7F2E9');
        sheet.getRange(2,1).setValue('ORDERS');
        sheet.getRange(2,1).setFontWeight('bold').setBackground('#F0EBE0').setFontColor('#888888');
        var oh = sheet.getRange(3,1,1,9);
        oh.setValues([['Logged at','Date','Order #','Channel','Qty','Unit Price','Total','Stock After','Notes']]);
        oh.setFontWeight('bold').setBackground('#0E0C0A').setFontColor('#F0C060');
        sheet.getRange(5,1).setValue('');  // spacer
        sheet.getRange(6,1).setValue('');  // spacer
        sheet.getRange(7,1).setValue('CONSIGNMENT');
        sheet.getRange(7,1).setFontWeight('bold').setBackground('#F0EBE0').setFontColor('#888888');
        var ch = sheet.getRange(8,1,1,9);
        ch.setValues([['Logged at','Date','Store','Event','Qty','Rate %','Amount Due','Notes','Status']]);
        ch.setFontWeight('bold').setBackground('#0E0C0A').setFontColor('#F0C060');
        sheet.setFrozenRows(3);
        ordersHeaderRow = 2;
      }
      // Find the orders header row (row after ORDERS label) and append below it
      // Orders data starts at row ordersHeaderRow+2 (header) and we insert before CONSIGNMENT
      var consRow = -1;
      for (var r2 = 1; r2 <= sheet.getLastRow(); r2++) {
        if (sheet.getRange(r2,1).getValue() === 'CONSIGNMENT') { consRow = r2; break; }
      }
      var insertRow = consRow > 0 ? consRow - 1 : sheet.getLastRow() + 1;
      // Find last data row in orders section
      var orderDataEnd = ordersHeaderRow + 2;
      for (var r3 = ordersHeaderRow + 2; r3 < (consRow > 0 ? consRow : sheet.getLastRow() + 1); r3++) {
        if (sheet.getRange(r3, 3).getValue() !== '') orderDataEnd = r3 + 1;
      }
      sheet.insertRowBefore(Math.max(orderDataEnd, ordersHeaderRow + 2));
      sheet.getRange(Math.max(orderDataEnd, ordersHeaderRow + 2), 1, 1, 9)
        .setValues([[timestamp, data.date, data.num, data.chan, data.qty, data.price, data.total, data.stockAfter, data.notes||'']]);
    }

    if (data.type === 'consignment') {
      var sheet2 = ss.getSheetByName(shortName);
      if (!sheet2) return; // should exist from order creation
      var consHeaderRow = -1;
      for (var r4 = 1; r4 <= sheet2.getLastRow(); r4++) {
        if (sheet2.getRange(r4,1).getValue() === 'CONSIGNMENT') { consHeaderRow = r4; break; }
      }
      if (consHeaderRow === -1) return;
      sheet2.appendRow([timestamp, data.date, data.store||'', data.event||'', data.qty, data.rate||'', data.amountDue||0, data.notes||'', data.status||'']);
    }

    var output = ContentService.createTextOutput(JSON.stringify({ok:true,ts:timestamp}));
    output.setMimeType(ContentService.MimeType.JSON);
    return output;

  } catch(err) {
    var errOut = ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message}));
    errOut.setMimeType(ContentService.MimeType.JSON);
    return errOut;
  }
}

// Test with GET so you can verify deployment in browser
